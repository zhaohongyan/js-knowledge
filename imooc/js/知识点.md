  ``` 
  // 手写bind函数
  Function.prototype.bind1 = function () {
    // 将参数拆解成数组
    const args = Array.prototype.slice.apply(arguments);

    // 获取this
    const t = args.shift();
    
    // fn1.bind(...)  中的fn1
    const self = this;

    return function () {
      return self.apply(t, args)
    }
  }
  ```
